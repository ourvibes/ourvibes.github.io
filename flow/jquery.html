    ajax: {
        ajaxPost: function (url, data, handler, dataType, async) {
          var dtd = $.Deferred(),
              data;
          $.ajax({
            url: url,
            data: data || {},
            type: 'POST',
            cache: false,
            async: async == undefined ? true : false,
            dataType: dataType || 'json'
          }).done(function (data) {
            data = handler ? handler(data) : data
            dtd.resolve(data)
          }).fail(function () {
            dtd.reject(data)
          });
          return dtd.promise()
        },
           
        ajaxGet: function (url, data, handler, dataType, async) {
          var dtd = $.Deferred(),
            data;
          $.ajax({
            url: url,
            data: data || {},
            type: 'GET',
            cache: false,
            async: async === undefined ? true : false,
            dataType: dataType || 'json'
          }).done(function (data) {
            data = handler ? handler(data) : data
            dtd.resolve(data)
          }).fail(function () {
            dtd.reject(data)
          });
          return dtd.promise()
        }
      },
      
      getUrlParam: function (name) {
        var reg = new RegExp('(^|&)' + name + '=([^&]*)(&|$)'), 
          r = window.location.search.substr(1).match(reg)
        if (r != null) {
          return decodeURIComponent(r[2])
        }
        return null
      },

      isEmpty: function (str) {
        str = $.trim(str)
        if (str == "" || str == null || str == undefined || str == "null" || str.length <= 0 ) return true
        return false

        zTree.hideNodes(hiddenNodes)

    return  (parseFloat(rs)*100) .toFixed(2)+ "%"

  var bootstrap = require('bootstrap')
  var dao = require("dao")
  var CountUp = require("countUp")
  var template = require('template')
  var utils = require('utils')
  
  var GLOBAL = {}
  
  var demo = {
    some:function() {
      Spin.add('#indexTopInfo')
      var params = {
      "style": GLOBAL.indexTopStyle
      }
      $.when(dao.methodsName(params)).done(function (data) {
       if(data.status) {
        var rs = data.data || {}
        var html = template('indexTopInfoTemplate',{"rs":rs})
        $('#indexTopInfo').html(html)
        _countUpTask()
      }
      Spin.remove('#indexTopInfo')
     })
    }
  }
  
  var _countUpTask = function(area) {
    setTimeout(function() {
      var options = {
        useEasing: true,
        useGrouping: true,
        separator: ',',
        decimal: '.',
      };
      var l = $(area ||"body").find(".countup");
      if( l.length<=0 ) return;
      var d = [2,2.1,2.2,2.3,2.4,2.5,2.6,2.7,2.8,2.9,3,3.1,3.2,3.4,3.5,3.6];
      l.each(function(i,e) {
        var  id = "countup_id_" + i , val = $(e).attr("count-val");
        $(e).attr("id",id);
        var countup = new CountUp(id, 0, val , 0, d[parseInt(Math.random()*17,10)] || 2.4 , options);
        !countup.error == true ?  countup.start() : $("#" + id).text(val);
      })
    }, 0)
  }

  var __main = function() {
    utils.executeFun(demo)
  }
  
  $(function() {
    __main()
    _countUpTask()
  })
